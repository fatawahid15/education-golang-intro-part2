package main

import (
	"fmt"
	"os"
)

func main() {

	defer fmt.Println("Deffered statement")

	fmt.Println("Starting the main function")

	//Exit with status code of 1
	os.Exit((1))

	//This wont ever be executed
	fmt.Println("End of main function")

	// ^^ Immediate exit without running any cleanup code, so the print below the os exit and the defer above wont ever be printed

}
