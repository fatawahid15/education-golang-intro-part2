package main

import (
	"fmt"
	"math/rand"
	"time"
)

func main() {

	source := rand.NewSource(time.Now().UnixNano())
	random := rand.New(source)

	// Generate a random number between 1 and 100
	target := random.Intn(100) + 1

	// Welcome
	fmt.Println("Welcome to the Guessing Game!")
	fmt.Println("chose number between 1-100")

	var guess int
	for {
		fmt.Println("Enter your guess: ")
		fmt.Scanln(&guess) // & <= address of the variable, the input in the cli will be putted in the variable address

		// if ELSE
		if guess == target {
			fmt.Println("gg u win")
			break
		} else if guess < target {
			fmt.Println("too low")
		} else {
			fmt.Println("too high")
		}
	}
}
