package main

import "fmt"

func main() {

	message := "Hello World"

	for i, v := range message {
		// fmt.Println(i, v)
		fmt.Printf("index: %d, Rune: %c\n", i, v)
		// ff you modify inside this loop the range won't able to print the changed value, because it works by copying the original value
	}

	for _, v := range message {
		// fmt.Println(i, v)
		fmt.Printf("Rune: %c\n", v)
		// ff you modify inside this loop the range won't able to print the changed value, because it works by copying the original value
	}

	// ^ this way kinda bit usable to prevent memory leak, well that's the whole point of a blank so less resource is wasted

}
